
环形队列最多有Maxsize-1个元素。

需要额外注意的是这里的DispQueue(),如果循环条件改为`f % Maxsize <= r`,当r处于`Maxsize-1` 的时候，f 永远不可能出循环, 所以这里循环条件可以写为`f % Maxsize != r`, 或者 `f % Maxsize < r` 。

除此之外，还需要特判一个CntQueue为0的情况。
DispQueue()改成这样就很好：
```cpp
void DispQueue(SqQueue* qu)
{
	if (qu->front == qu->rear) return;
	int f = (qu->front + 1) % Maxsize, r = qu->rear;

	while (f % Maxsize < r)
	{
		cout << qu->data[f] << " ";
		f = (f + 1) % Maxsize;
	}
	cout << qu->data[f];
	cout << endl;
}
```

初始头文件：
```cpp
#pragma once

#include<iostream>
#include<malloc.h>
#define Elemtype int

using namespace std;
const int Maxsize = 10;

typedef struct
{
	Elemtype data[Maxsize];
	int front, rear;
}SqQueue;

//初始化队列
void InitQueue(SqQueue*& qu)
{
	qu = (SqQueue*)malloc(sizeof(SqQueue));
	qu->front = qu->rear = 0;
}

//销毁队列
void DestroyQueue(SqQueue* &qu)
{
	free(qu);
}

//判断队列是否为空
bool EmptyQueue(SqQueue*& qu)
{
	return (qu->front == qu->rear);
}

//进队
bool enQueue(SqQueue*& qu, Elemtype &e)
{
	if ((qu->rear + 1) % Maxsize == qu->front) return false;

	qu->rear = (qu->rear + 1) % Maxsize;
	qu->data[qu->rear] = e;
	return true;
}

//出队
bool deQueue(SqQueue*& qu, Elemtype &e)
{
	if (qu->front == qu->rear) return false;

	qu->front = (qu->front + 1) % Maxsize;
	e = qu->data[qu->front];
	return true;
}

//计算环形队列中元素的个数
int CntQueue(SqQueue*& qu)
{
	return (qu->rear - qu->front + Maxsize) % Maxsize;
}

//从队头到队尾输出队列
/*注释掉的方法不可以了
void DispQueue(SqQueue* qu)
{
	for (int i = qu->front + 1; i <= qu->rear; i = (i+1)%Maxsize)
		cout << qu->data[i] << " ";
	cout << endl;
}*/
void DispQueue(SqQueue* qu)
{
	int f = (qu->front + 1) % Maxsize, r = qu->rear;
	while (f % Maxsize != r)
	{
		cout << qu->data[f] << " ";
		f = (f + 1) % Maxsize;
	}
	cout << qu->data[f];
	cout << endl;
}
```

源文件
```cpp
#include"顺序表实现环形队列.h"
#include<iostream>
#include<malloc.h>

using namespace std;

int main()
{
	int op = 100;
	int e;

	SqQueue* queue;
	InitQueue(queue);
	cout << "队列初始化完毕" << endl;
	cout << "enQueue输入1，deQueue输入2." << endl;

	while (cout << "请输入option:" << endl, cin >> op && op)
	{
		if (op == 1)
		{
			cin >> e;
			enQueue(queue, e);
			cout << "当前队列输出为：";
			DispQueue(queue);
			cout << "当前队列中有" << CntQueue(queue) << "个元素" << endl;
		}
		else if (op == 2)
		{
			deQueue(queue, e);
			cout << "出队元素为：" << e << endl;
			cout << "当前队列输出为：";
			DispQueue(queue);
			cout << "当前队列中有" << CntQueue(queue) << "个元素" << endl;
		}
	}

	if (EmptyQueue(queue))
		cout << "队列空" << endl;

	DestroyQueue(queue);
	cout << "已销毁队列" << endl;
}
```
