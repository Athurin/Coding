[203. 移除链表元素 - 力扣（LeetCode）](https://leetcode.cn/problems/remove-linked-list-elements/)

AC1: 不使用虚拟节点，删除头节点和非头节点的时候分类讨论
很关键的是两个判断语句是`while` 因为可能会出现对各删除目标。
还有`else cur = cur->next;` 里面的`else` 非常关键，因为删除完结点之后，`cur`的下一个新的节点还没有出现，所以`cur`不能向下移动。
时间复杂度O(n)
空间复杂度O(1)
```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution 
{
public:
    //不使用虚拟节点，删除头节点和非头节点的时候分类讨论
    ListNode* removeElements(ListNode* head, int val) 
    {
        ListNode* tmp = nullptr;

        //头节点
        while(head && head->val == val)
        {
            tmp = head;
            head = head->next;
            delete tmp;
        }

        //非头节点
        ListNode* cur = head;
        while(cur && cur->next)
        {
            if(cur->next->val == val)
            {
                tmp = cur->next;
                cur->next = tmp->next;
                delete tmp;
            }
            else cur = cur->next;
        }
        return head;
    }
};
```

AC2: 添加虚拟头节点，不再分类讨论
时间复杂度O(n)
空间复杂度O(1)
```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution 
{
public:
    ListNode* removeElements(ListNode* head, int val)
    {
        ListNode* dummyhead = new ListNode(0,head);
        ListNode* cur = dummyhead;
        ListNode* tmp = nullptr;
        while(cur->next)
        {
            if(cur->next->val == val)
            {
                tmp = cur->next;
                cur->next = tmp->next;
                delete tmp, tmp = nullptr;
            }
            else cur = cur->next;
        }
        head = dummyhead->next;
        delete dummyhead;
        return head;
    }
};
```

AC3: 递归   不推荐
时间复杂度O(n) ，虽然是线性的，但是依然会大于方法一二
空间复杂度O(n)
```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution 
{
public:
    ListNode* removeElements(ListNode* head, int val) 
    {
        if(!head)
            return head;

        // 递归处理
        if (head->val == val) 
        {
            ListNode* newHead = removeElements(head->next, val);
            delete head;
            return newHead;
        } 
        else 
        {
            head->next = removeElements(head->next, val);
            return head;
        }
    }
};
```
