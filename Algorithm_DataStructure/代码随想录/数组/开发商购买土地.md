[44. 开发商购买土地（第五期模拟笔试）](https://kamacoder.com/problempage.php?pid=1044)

很简单的二维前缀和：
```cpp
#include <iostream>
#include <cmath>

using namespace std;
const int N = 110;
int n, m;
int x;
int s[N][N];

void solve()
{
    int res = INT32_MAX;

    for(int i=1; i<=n; i++)
    {
        for(int j=1; j<=m; j++)
        {
            cin >> x;
            s[i][j] = s[i-1][j] + s[i][j-1] - s[i-1][j-1] + x;
        }
    }

    for(int i=1; i<n; i++)
    {
        x = abs(s[i][m] - (s[n][m] - s[i][m]));
        res = min(res, x);
    }

    for(int i=1; i<m; i++)
    {
        x = abs(s[n][i] - (s[n][m] - s[n][i]));
        res = min(res, x);
    }

    cout << res;
}

int main()
{
    cin >> n >> m;
    solve();
    return 0;
}
```
