#include<iostream>

using namespace std;

const int N =1e5+10;
int p[N];

int find(int x)
{
    if(p[x]!=x) p[x]=find(p[x]);

    //若是return x，表示回溯的每一步都返回的是父节点编号而不是祖宗节点的编号
    //return p[x]，表示递归至祖宗节点后就把祖宗节点的编号返回
    //每一次回溯，都把原来的父结点用祖宗节点覆盖，然后再return
    //于是在这条路径上的每一个结点都指向了祖宗节点，从而起到了压缩路径的目的
    return p[x];
}

int main()
{
    int n,m;
    cin>>n>>m;

    //最开始每一个元素都是一个独立的集合
    //于是元素 i 的祖宗节点的编号就是i 
    for(int i=1;i<=n;i++) p[i]=i;

    while(m--)
    {
        char op;
        int a,b;
        cin>>op>>a>>b;

        if(op=='M') p[find(a)]=find(b);
        else if(op=='Q')
        {
            if(find(a)==find(b)) printf("Yes\n");
            else printf("No\n");
        }
    }

    return 0;
}

#include<iostream>

using namespace std;

const int N=1e5+10;
int p[N],cnt[N];

int find(int x)
{
    if(p[x]!=x) p[x]=find(p[x]);

    return p[x];
}

int main()
{
    int n,m;
    cin>>n>>m;

    for(int i=1;i<=n;i++)
    {
        p[i]=i;
        cnt[i]=1;
    }

    while(m--)
    {
        //字符串的输入都用string
        string op;
        int a,b;
        cin>>op;

        if(op=="C")
        {
            cin>>a>>b;
            a=find(a);b=find(b);
            if(a!=b)
            {
                p[a]=b;
                cnt[b]+=cnt[a];
            }
        }

        else if(op=="Q1")
        {
            cin>>a>>b;
            a=find(a);b=find(b);

            if(a==b) printf("Yes\n");
            else printf("No\n");
        }

        else
        {
            cin>>a;
            a=find(a);

            cout<<cnt[a]<<endl;
        }

    }

    return 0;
}
