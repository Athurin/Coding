
## 单链表

[826. 单链表 - AcWing题库](https://www.acwing.com/problem/content/828/)

```cpp
//法一：用head指针
//头插和在中间插操作不一样，要特判
//头删和在中间删除操作不一样
#include<iostream>

using namespace std;

const int N=1e5+10;

//head表示头结点的下标；
//e[N]存放的是第N+1个结点的值；
//ne[N]存放的第N+2个结点的下标；
//idx表示当前的下标；
int head,e[N],ne[N],idx;

//初始化
void init()
{
    head=-1;//最开始链表为空
    idx=0;
}

//在头结点的位置插入值
void head_add(int x)
{
    e[idx]=x;
    ne[idx]=head;//若head=-1,则ne[idx]=-1,总是满足尾结点的ne值为-1；
    head=idx;
    idx++;
}

//删除下标为k的结点的后面的一个结点
void remove(int k)
{
    ne[k]=ne[ne[k]];
}

//在下表为k的结点的后面的位置插入值为x的结点
void insert_elm(int k,int x)
{
    e[idx]=x;
    ne[idx]=ne[k];
    ne[k]=idx;
    idx++;
}

int main()
{
    int m;
    cin>>m;
    
    init();
    
    while(m--)
    {
        char op;
        int k,x;
        
        cin>>op;
        //字符的输入不用scanf(),因为scanf会读入回车或者空格，从而导致输入操作无法正常进行
        
        if(op=='H')
        {
            scanf("%d",&x);
            head_add(x);
        }
        else if(op=='D')
        {
            scanf("%d",&k);
            if(!k) head=ne[head];//根据题意会有改变
            else remove(k-1);//第K个结点的下标为k-1；
        }
        else
        {
            scanf("%d%d",&k,&x);
            insert_elm(k-1,x);
        }
    }
    
    for(int i=head;i!=-1;i=ne[i]) cout<<e[i]<<' ';
    cout<<endl;
    
    return 0;
}
```

```cpp
//法二：占用ne[0]做头结点
//不管是插入还是删除都不用做头节点的特判
//对应的下标就是插入的顺序，不用减一
#include<iostream>

using namespace std;

const int N =1e5+10;
int idx,e[N],ne[N];

void init()
{
    ne[0]=-1;
    idx=1;
}

void remove(int k)
{
    ne[k]=ne[ne[k]];
}

void add(int k,int x)
{
    e[idx]=x;
    ne[idx]=ne[k];
    ne[k]=idx;
    idx++;
}

int main()
{
    int m;
    cin>>m;
    
    init();
    
    while(m--)
    {
        char op;
        cin>>op;
        
        if(op=='H')
        {
            int x;
            cin>>x;
            add(0,x);
        }
        else if(op=='D')
        {
            int k;
            cin>>k;
            remove(k);
        }
        else if(op=='I')
        {
            int k,x;
            cin>>k>>x;
            add(k,x);
        }
    }
    
    for(int i=ne[0];i!=-1;i=ne[i]) cout<<e[i]<<' ';
    cout<<endl;
    
    return 0;    
}
```

## 双链表

[827. 双链表 - AcWing题库](https://www.acwing.com/problem/content/829/)

```cpp
#include<iostream>

using namespace std;

const int N = 1e5+10;
int idx,e[N],l[N],r[N];

//初始化双链表
//0,1头尾结点不算有效结点
void init()
{
    r[0]=1;
    l[1]=0;
    idx=2;
}

//在下标为a的结点的右边插入一个数
void insert(int a,int x)
{
    e[idx]=x;
    r[idx]=r[a];
    l[idx]=a;
    l[r[a]]=idx;
    r[a]=idx;
    idx++;
}

//删除下表为a的结点
void remove(int a)
{
    r[l[a]]=r[a];
    l[r[a]]=l[a];

}

int main()
{
    int m;
    cin>>m;

    init();

    while(m--)
    {
        //字符串输入时，scanf()会有很多限制，cin又没这么快
        //可以用ios::sync_with_stdio(false) 来提速cin
        //这句代码目的是 让输入的值不在缓冲区停留，提高cin的速度
        ios::sync_with_stdio(false);


        string op;
        cin>>op;
        int k,x;

        if(op=="L")
        {
            cin>>x;
            insert(0,x);
        }
        else if(op=="R")
        {
            cin>>x;
            //尾结点左边的结点的右边进行插入操作
            //尾节点左边的第一个结点是最后一个有效结点
            insert(l[1],x);
        }
        else if(op=="D")
        {
            cin>>k;
            //第一个优效结点的下表为2
            //第k 个结点的下标为K+1
            remove(k+1);
        }
        else if(op=="IL")
        {
            cin>>k>>x;
            insert(l[k+1],x);
        }
        else if(op=="IR")
        {
            cin>>k>>x;
            insert(k+1,x);
        }
    }

    for(int i=r[0];i!=1;i=r[i]) cout<<e[i]<<' ';
    cout<<endl;

    return 0;
}
```
